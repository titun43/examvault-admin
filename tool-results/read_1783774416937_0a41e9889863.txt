   178→      // ---- Name-based parent resolution ----
   179→      // Each row can use EITHER `subjectId` (explicit Firestore doc id —
   180→      // backward compatible) OR `subjectName` (resolved to id via Firestore
   181→      // lookup, optionally disambiguated by `categoryName`). If both are
   182→      // present, subjectId wins. If neither is present, the row is skipped.
   183→      //
   184→      // We ALSO resolve categoryName -> categoryId here so the row keeps a
   185→      // denormalized categoryId (the Test model stores subjectId only, but

   187→      // -----------------------------------------------------------------
   188→      const skipList: { row: number; reason: string }[] = [];

   191→      const subjIdCache = new Map<string, string | null>();

   201→        // Cache key for subject: name + '||' + categoryName (disambiguator).
   202→        const subjKey = `${subjName.toLowerCase()}||${catName.toLowerCase()}`;
   203→

   205→          if (subjIdCache.has(subjKey)) {
   206→            subjectId = subjIdCache.get(subjKey) ?? '';
   207→          } else {
   208→            const r = await resolveSubjectIdByName(subjName, catName || undefined);
   209→            subjIdCache.set(subjKey, r.id);
   210→            subjectId = r.id ?? '';

   213→            skipList.push({
   214→              row: rowNo,
   215→              reason: `subject "${subjName}"${catName ? ` (in ${catName})` : ''} not found`,
   216→            });
   217→            continue;
   218→          }
   219→        }
   220→        if (!subjectId) {
   221→          skipList.push({ row: rowNo, reason: 'no subjectId or subjectName provided' });
   222→          continue;
   223→        }

   226→        payload.subjectId = subjectId;
   227→        delete payload.subjectName;

   229→        // (Skip if the row already had categoryId — explicit wins.)

   232→            const cid = catIdCache.get(catName.toLowerCase());

   234→          } else {
   235→            const c = await resolveCategoryIdByName(catName);
   236→            catIdCache.set(catName.toLowerCase(), c.id);
   237→            if (c.id) payload.categoryId = c.id;
   238→          }

   241→        if (!payload.updatedAt) payload.updatedAt = serverTimestamp();
   242→        resolved.push(payload);
   243→      }

   311→      }
   312→      setBulkOpen(false);

   315→      toast.error(err?.message || 'Bulk import failed');
